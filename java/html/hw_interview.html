华为Java面试题

  1. 第一题

    public class Test1 {
    public static void changeStr(String str) {
	str = "welcome";
    }
    public static void main(String[] argv) {
	String str = "1234";
	changeStr( str );
	System.out.println( str );
    }
}

    答案: 1234

    public class Test2 {

       System.out.print(c);
       return true;

    public static void main(String[] argv) {
       int i = 0;
       //for(65;88&&(i<2);67)
       for (foo('A'); foo('B') && (i < 2); foo('C')) {
           i++;
           foo('D');
       }
    }
}
