判断元素是否是某个元素的子元素

// 判断当前元素是否是某个元素的子元素
function isChild (queue, target) {
	var node, found = false;
	
	while(!found && queue.length) {
		node = queue.shift();
		
		if (node === target) {
			found = true;
		} else if (node.childNodes.length) {
			Array.prototype.push.apply(queue, node.childNodes);
		}
	}
	
	return found;
}
